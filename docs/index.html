example

const fs = require('fs')
const { PEM, ASN1 } = require('@fidm/asn1')

const pems = PEM.parse(fs.readFileSync('./test/cert/github.crt'))
const asn1 = ASN1.fromDER(pems[0].body)
console.log(asn1)
